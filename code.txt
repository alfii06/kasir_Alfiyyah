// Misalnya, menggunakan PDO untuk koneksi ke database
$servername = "localhost";
$username = "root";
$password = "";
$dbname = "pos_alfi";

try {
    $pdo = new PDO("mysql:host=$servername;dbname=$dbname", $username, $password);
    // Atur mode error ke Exception
    $pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);

    // Pastikan data yang diterima dari formulir sesuai dengan struktur tabel database Anda
    $pelanggan_id = $_POST['pelangganId'];
    $toko_id = $_POST['tokoId'];
    $nama_pelanggan = $_POST['namaPelanggan'];
    $alamat = $_POST['alamat'];
    $no_hp = $_POST['noHp'];

    // Contoh query untuk menyimpan perubahan
    $query = "UPDATE pelanggan SET toko_id = :toko_id, nama_pelanggan = :nama_pelanggan, alamat = :alamat, no_hp = :no_hp WHERE pelanggan_id = :pelanggan_id";

    // Persiapkan statement
    $stmt = $pdo->prepare($query);

    // Bind parameter
    $stmt->bindParam(':toko_id', $toko_id);
    $stmt->bindParam(':nama_pelanggan', $nama_pelanggan);
    $stmt->bindParam(':alamat', $alamat);
    $stmt->bindParam(':no_hp', $no_hp);
    $stmt->bindParam(':pelanggan_id', $pelanggan_id);

    // Eksekusi statement
    $stmt->execute();

    // Redirect ke halaman data_pelanggan.php setelah berhasil
    header("Location: data_pelanggan.php");
    exit();
} catch (PDOException $e) {
    echo "Error: " . $e->getMessage();
}

// Tutup koneksi
$pdo = null;
?>
